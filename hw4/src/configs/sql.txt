DROP TABLE IF EXISTS `SHEET`;
CREATE TABLE SHEET(
	sheet_id INT(10) UNSIGNED AUTO_INCREMENT,
	sheet_name VARCHAR(20) NOT NULL,
	sheet_data VARCHAR(5000) NOT NULL,
	UNIQUE(sheet_name),
	PRIMARY KEY(sheet_id));
DROP TABLE IF EXISTS `SHEET_CODES`;
CREATE TABLE SHEET_CODES(
	sheet_id INT(10) UNSIGNED NOT NULL,
	hash_code VARCHAR(20) NOT NULL,
	code_type VARCHAR(20) NOT NULL);
